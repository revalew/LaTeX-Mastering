\begin{align*}
    \int_0^1\frac{(x+1)(x-2)}{x-3}\,dx&=\int_0^1\frac{x^2-x-2}{x-3}\,dx\\
    &=\int_0^1x+2+\frac{4}{x-3}\,dx\\
    &=\bigg[\frac{1}{2}x^2+2x+4\ln|x-3|\bigg]_0^1\\
    &=\frac{5}{2}+4\ln(2/3).
\end{align*}
